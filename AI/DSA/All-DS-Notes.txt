from collections import deque

def bfs(root):

    q = deque([root])

    while q:

        node = q.popleft()

        print(node.val)

        if node.left:
            q.append(node.left)

        if node.right:
            q.append(node.right)

15. BINARY SEARCH

Pattern most frequently asked.

When to Use

Problem contains:

Sorted Array
Sorted Data
Search Space
Example
def binary_search(nums,target):

    left = 0
    right = len(nums)-1

    while left <= right:

        mid = (left+right)//2

        if nums[mid] == target:
            return mid

        elif nums[mid] < target:
            left = mid+1

        else:
            right = mid-1

    return -1
16. SLIDING WINDOW
Pattern

Subarray / Substring

Keywords:

Longest
Smallest
Maximum
Minimum
Continuous
Substring
Subarray
Example

Maximum sum size K

def max_sum(nums,k):

    window = sum(nums[:k])

    result = window

    for i in range(k,len(nums)):

        window += nums[i]
        window -= nums[i-k]

        result = max(result,window)

    return result
